## Purpose

`SlipPopUp` provides a playlist collection manager where users can:

- Create named "slips" (mini-playlists)
- Add/remove current song to slips
- Search slips with fuzzy matching
- View slip contents
- Load slips as playlists
- Persist slips to localStorage

```tsx
  threshold: 0.3,      // Fuzzy matching threshold

  ? fuse.search(searchQuery).map((result) => result.item)
```

### Remove Song from Slip

```tsx
const deleteSong = (song: { title: string; id: string }, index: number) => {
  const newSlips = [...slips];
  const songsCopy = [...newSlips[index].songs];
  const song_index = songsCopy.findIndex((s) => s.id === song.id);
  
  songsCopy.splice(song_index, 1);
  newSlips[index].songs = songsCopy;
  setSlips(newSlips);
};
```

### CheckMark Component

```tsx
    const currentSong = appContext.loadedCard.songs[appContext.currentTrack];
    const isAlreadyInSlip = slip.songs.some(
      (song) => song.id === currentSong?.id
    );

    } else {
      addSong(currentSong, index);

  }}
```

## Click Outside Handler

Closes popup when clicking outside:

```tsx
const popupRef = useRef<HTMLDivElement>(null);

useEffect(() => {
  const handleClickOutside = (e: MouseEvent) => {
    if (popupRef.current && !popupRef.current.contains(e.target as Node)) {
      setShowSlipPopUp(false);
    }
  };
  
  if (showSlipPopUp) {
    document.addEventListener("mousedown", handleClickOutside);
  }
  
  return () => {
    document.removeEventListener("mousedown", handleClickOutside);
  };
}, [showSlipPopUp]);
```

## Integration Example

```tsx
import SlipPopUp from '@/components/SlipPopUp/SlipPopUp';
import { AnimatePresence } from 'framer-motion';
import { useState } from 'react';

function PlayerComponent() {
  const [showSlipPopUp, setShowSlipPopUp] = useState(false);
  
  return (
    <>
      <button onClick={() => setShowSlipPopUp(true)}>Open Slips</button>
      
      <AnimatePresence>
        {showSlipPopUp && (
          <SlipPopUp
            showSlipPopUp={showSlipPopUp}
            setShowSlipPopUp={setShowSlipPopUp}
          />
        )}
      </AnimatePresence>
    </>
  );
}
```
